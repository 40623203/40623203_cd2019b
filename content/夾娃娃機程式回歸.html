夾娃娃機程式回歸

function sysCall_init()

slider=sim.getObjectHandle("y_joint")
sliding = -0.145
slider_1=sim.getObjectHandle("x_joint")
sliding_1 = -0.195
slider_2 = sim.getObjectHandle("z_joint")
sliding_2 = 0
slider_3=sim.getObjectHandle("slider_joint")
sliding_3 = 0
Sensor=sim.getObjectHandle("sensor")
C = 0

end

function sysCall_actuation()

sim.setJointTargetPosition(slider, sliding)
sim.setJointTargetPosition(slider_1, sliding_1)
sim.setJointTargetPosition(slider_2, sliding_2)
sim.setJointTargetPosition(slider_3, sliding_3)

---return
A=sim.readProximitySensor(Sensor)
if (A>0) then C=sim.getSimulationTime() end
if (C > 0) then
    sliding_2 = sliding_2 +0.001
    if (sliding_2 >= 0) then sliding_1 = sliding_1 -0.01 end
    if (sliding_1 <= -0.195) then sliding = sliding -0.01 end
    if (sliding <= -0.145) then sliding_3 = -1 end
end
if (sliding_3 <= 0.01) then C=0 end
---

message,auxiliaryData=sim.getSimulatorMessage()
        while message ~= -1 do
    -- front and behind ( key : w and s )
            if (message==sim.message_keypress) then
                if (auxiliaryData[1]==119) then
                    -- front key
                     sliding = sliding + 0.01
    sim.addStatusbarMessage('sliding:'..sliding) 
                end
                if (auxiliaryData[1]==115) then
                    -- behind key
                     sliding = sliding - 0.01
    sim.addStatusbarMessage('sliding:'..sliding)
                end
                if (sliding < -0.145) then
                    sliding = -0.145
                elseif (sliding > 0.145) then
                    sliding = 0.145
                end
    -- left and right ( key : a and d )
                if (auxiliaryData[1]==100) then
                    -- right key
                     sliding_1 = sliding_1 + 0.01
    sim.addStatusbarMessage('sliding_1:'..sliding_1) 
                end
                if (auxiliaryData[1]==97) then
                    -- left key
                     sliding_1 = sliding_1 - 0.01
    sim.addStatusbarMessage('sliding_1:'..sliding_1)
                end
                if (sliding_1 < -0.195) then
                    sliding_1 = -0.195
                elseif (sliding_1 > 0.195) then
                    sliding_1 = 0.195
                end
-- up and down ( key : up and down )
                if (auxiliaryData[1]==2007) then
                    -- up key
                     sliding_2 = sliding_2 + 0.01
    sim.addStatusbarMessage('sliding_2:'..sliding_2) 
                end
                if (auxiliaryData[1]==2008) then
                    -- down key
                     sliding_2 = sliding_2 - 0.01
    sim.addStatusbarMessage('sliding_2:'..sliding_2)
                end
                if (sliding_2 < -0.26) then
                    sliding_2 = -0.26
                elseif (sliding_2 > 0) then
                    sliding_2 = 0
                end
    -- close ( key : space )
                if (auxiliaryData[1]==32) then
                    -- close key
                     sliding_3 = sliding_3 + 0.01
    sim.addStatusbarMessage('sliding_3:'..sliding_3) 
                end
                if (sliding_3 < -0) then
                    sliding_3 = -0
                elseif (sliding_3 > 0.0185) then
                    sliding_3 = 0.0185
                end

            end
            message,auxiliaryData=sim.getSimulatorMessage()
        end

end
