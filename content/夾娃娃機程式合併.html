夾娃娃機程式合併

-- model in Solvespace 500 mm = 5000 mm in V-rep
--[[
Simulation is 10 times of realistic environment
floor in Solvespace 2.5 m x 2.5 m = 25 m x 25 m in V-rep
ball is in Solivespace 1g (0.001) = 0.01 kg in V-rep
hammer is in Solvespace 0.1 kg (100g) = 1kg in V-rep (0.1 for Inertia)
]]

threadFunction=function()
    while sim.getSimulationState()~=sim.simulation_advancing_abouttostop do
        -- Read the keyboard messages (make sure the focus is on the main window, scene view):
        message,auxiliaryData=sim.getSimulatorMessage()
        while message~=-1 do
            if (message==sim.message_keypress) then
-- front and behind ( key : w and s)
                if (auxiliaryData[1]==119) then
                    -- front key
                     sliding = sliding + 0.01
sim.addStatusbarMessage('sliding:'..sliding) 
                end
                if (auxiliaryData[1]==115) then
                    -- behind key
                     sliding = sliding - 0.01
sim.addStatusbarMessage('sliding:'..sliding)
                end
                if (sliding < -0.145) then
                    sliding = -0.145
                elseif (sliding > 0.145) then
                    sliding = 0.145
                end
-- left and right ( key : a and d )
                if (auxiliaryData[1]==97) then
                    -- left key
                     sliding_1 = sliding_1 + 0.01
sim.addStatusbarMessage('sliding_1:'..sliding_1) 
                end
                if (auxiliaryData[1]==100) then
                    -- right key
                     sliding_1 = sliding_1 - 0.01
sim.addStatusbarMessage('sliding_1:'..sliding_1)
                end
                if (sliding_1 < -0.195) then
                    sliding_1 = -0.195
                elseif (sliding_1 > 0.195) then
                    sliding_1 = 0.195
                end
-- up and down ( key : up and down )
                if (auxiliaryData[1]==2007) then
                    -- up key
                     sliding_2 = sliding_2 + 0.01
sim.addStatusbarMessage('sliding_2:'..sliding_2) 
                end
                if (auxiliaryData[1]==2008) then
                    -- down key
                     sliding_2 = sliding_2 - 0.01
sim.addStatusbarMessage('sliding_2:'..sliding_2)
                end
                if (sliding_2 < -0.26) then
                    sliding_2 = -0.26
                elseif (sliding_2 > 0) then
                    sliding_2 = 0
                end
-- close and open ( key : left and right )
                if (auxiliaryData[1]==2009) then
                    -- close key
                     sliding_3 = sliding_3 + 0.01
sim.addStatusbarMessage('sliding_3:'..sliding_3) 
                end
                if (auxiliaryData[1]==2010) then
                    -- open key
                     sliding_3 = sliding_3 - 0.01
sim.addStatusbarMessage('sliding_3:'..sliding_3)
                end
                if (sliding_3 < -0) then
                    sliding_3 = -0
                elseif (sliding_3 > 0.0185) then
                    sliding_3 = 0.0185
                end
            end
            message,auxiliaryData=sim.getSimulatorMessage()
        end
        sim.setJointTargetPosition(slider, sliding)
        sim.setJointTargetPosition(slider_1, sliding_1)
        sim.setJointTargetPosition(slider_2, sliding_2)
        sim.setJointTargetPosition(slider_3, sliding_3)
        -- Since this script is threaded, don't waste time here:
        sim.switchThread() -- Resume the script at next simulation loop start
    end
end
-- Put some initialization code here:
-- Retrieving of some handles and setting of some initial values:
slider=sim.getObjectHandle("y_joint")
sliding = -0.145
slider_position=sim.getJointPosition(slider, -1)
slider_1=sim.getObjectHandle("x_joint")
sliding_1 = -0.195
slider_position_1=sim.getJointPosition(slider_1, -1)
slider_2 = sim.getObjectHandle("z_joint")
sliding_2 = 0
slider_position_2=sim.getJointPosition(slider_2, -1)
slider_3=sim.getObjectHandle("slider_joint")
sliding_3 = 0
slider_position_3=sim.getJointPosition(slider_3, -1)
-- Here we execute the regular thread code:
res,err=xpcall(threadFunction,function(err) return debug.traceback(err) end)
if not res then
    sim.addStatusbarMessage('Lua runtime error: '..err)
end
